CREATE TABLE `wisdom_park`.`details` ( `id` INT NOT NULL AUTO_INCREMENT , `section` VARCHAR(255) NOT NULL , `description` LONGTEXT NOT NULL , `answer` LONGTEXT NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;

CREATE TABLE `wisdom_park`.`users` ( `id` INT NOT NULL AUTO_INCREMENT , `name` VARCHAR(255) NOT NULL , `username` VARCHAR(255) NOT NULL , `password` VARCHAR(255) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
